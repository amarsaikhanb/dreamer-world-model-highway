Project Overview



This notebook implements a Dreamer V1-style World Model from scratch in PyTorch,
applied to the highway-v0 environment from Gymnasium's Highway-Env package.

The agent's objective is to drive at high speed while avoiding collisions with
neighboring vehicles. The world model learns to predict future states from images
and enables the actor-critic policy to train "in imagination."

Architecture Components:
1. Encoder — CNN that maps RGB frames -> latent representation
2. RSSM — Recurrent State-Space Model for temporal dynamics
3. Reward Model — MLP predicting rewards from latent states
4. Decoder — Transposed CNN reconstructing frames from latent states
5. Actor (Policy) — MLP outputting continuous actions
6. Critic (Value) — MLP estimating state values

def make_env():
    """Create and configure the highway-v0 environment.

    Returns a Gymnasium environment configured for continuous control
    with RGB array rendering for frame-based world model training.
    """
    env = gym.make('highway-v0', render_mode='rgb_array')

    # Configure for continuous actions and 4 lanes
    env.unwrapped.configure({
        "action": {
            "type": "ContinuousAction",        # Two continuous floats: steering & acceleration
        },
        "lanes_count": NUM_LANES,              # 4 lanes for richer dynamics
        "vehicles_count": 20,                  # Number of surrounding vehicles
        "duration": 40,                        # Episode length in env steps
        "simulation_frequency": 15,            # Physics simulation frequency
        "policy_frequency": 5,                 # Decision frequency
        "observation": {
            "type": "GrayscaleObservation",
            "observation_shape": (IMG_WIDTH, IMG_HEIGHT),
            "stack_size": 1,
            "weights": [0.2989, 0.587, 0.114],
        },
        "screen_width": 600,
        "screen_height": 150,
        "reward_speed_range": [20, 30],        # Reward range for speed
        "collision_reward": -1.0,              # Penalty for collision
        "high_speed_reward": 0.4,              # Reward for high speed
        "right_lane_reward": 0.0,              # No lane preference
    })

    env.reset()
    return env


def preprocess_frame(frame):
    """Preprocess a raw environment frame for the world model.

    Resizes to (IMG_HEIGHT, IMG_WIDTH), normalizes to [0, 1], and
    converts to CHW tensor format expected by PyTorch CNNs.

    Args:
        frame: Raw RGB frame from env.render(), shape (H, W, 3)

    Returns:
        Tensor of shape (3, IMG_HEIGHT, IMG_WIDTH) with values in [0, 1]
    """
    # Resize frame to target dimensions
    frame_resized = cv2.resize(frame, (IMG_WIDTH, IMG_HEIGHT),
                                interpolation=cv2.INTER_AREA)
    # Normalize to [0, 1] and convert HWC -> CHW
    frame_tensor = torch.FloatTensor(frame_resized).permute(2, 0, 1) / 255.0
    return frame_tensor

4. Replay Buffer



The replay buffer stores transition tuples (frame, action, reward, done, next_frame).
During training, we sample contiguous sequences of length SEQ_LEN for the RSSM,
which requires temporal coherence (unlike standard RL replay buffers that sample i.i.d.).
In [ ]:
class ReplayBuffer:
    """Episode-based replay buffer for sequence sampling.

    Stores full episodes and samples contiguous sub-sequences for training
    the RSSM-based world model, which needs temporal structure.
    Also tracks running reward statistics for normalization.

    Attributes:
        episodes: List of episode data dictionaries
        capacity: Maximum number of total transitions to store
    """

    def __init__(self, capacity=BUFFER_CAPACITY):
        self.episodes = []
        self.capacity = capacity
        self._total_transitions = 0
        self._all_rewards = []  # Track all rewards for statistics

    @property
    def reward_stats(self):
        """Return (mean, std) of all observed rewards for normalization."""
        if len(self._all_rewards) < 2:
            return 0.0, 1.0
        arr = np.array(self._all_rewards)
        return float(arr.mean()), float(max(arr.std(), 1e-6))

    def add_episode(self, frames, actions, rewards, dones):
        """Add a complete episode to the buffer.

        Args:
            frames: List of preprocessed frame tensors (T+1 frames)
            actions: List of action arrays (T actions)
            rewards: List of reward floats (T rewards)
            dones: List of done booleans (T dones)
        """
        episode = {
            'frames': torch.stack(frames),           # (T+1, C, H, W)
            'actions': torch.FloatTensor(np.array(actions)),  # (T, action_dim)
            'rewards': torch.FloatTensor(rewards),   # (T,)
            'dones': torch.FloatTensor(dones),       # (T,)
        }
        self.episodes.append(episode)
        self._total_transitions += len(rewards)
        self._all_rewards.extend(rewards)

        # Remove oldest episodes if over capacity
        while self._total_transitions > self.capacity and len(self.episodes) > 1:
            removed = self.episodes.pop(0)
            self._total_transitions -= len(removed['rewards'])

    def sample_sequences(self, batch_size, seq_len):
        """Sample random contiguous sequences from stored episodes.

        Rewards are normalized using running statistics to stabilize
        the reward model and value function training.

        Args:
            batch_size: Number of sequences to sample
            seq_len: Length of each sequence

        Returns:
            Dictionary with keys 'frames', 'actions', 'rewards', 'rewards_raw', 'dones'
        """
        valid_episodes = [ep for ep in self.episodes
                          if len(ep['rewards']) >= seq_len]

        if not valid_episodes:
            raise ValueError(f"No episodes with length >= {seq_len}")

        batch_frames = []
        batch_actions = []
        batch_rewards = []
        batch_dones = []

        for _ in range(batch_size):
            ep = random.choice(valid_episodes)
            max_start = len(ep['rewards']) - seq_len
            start = random.randint(0, max_start)

            batch_frames.append(ep['frames'][start:start + seq_len + 1])
            batch_actions.append(ep['actions'][start:start + seq_len])
            batch_rewards.append(ep['rewards'][start:start + seq_len])
            batch_dones.append(ep['dones'][start:start + seq_len])

        # Normalize rewards using running stats
        r_mean, r_std = self.reward_stats
        rewards_tensor = torch.stack(batch_rewards)
        rewards_normalized = (rewards_tensor - r_mean) / r_std

        return {
            'frames': torch.stack(batch_frames).to(device),
            'actions': torch.stack(batch_actions).to(device),
            'rewards': rewards_normalized.to(device),         # normalized
            'rewards_raw': rewards_tensor.to(device),         # original scale
            'dones': torch.stack(batch_dones).to(device),
        }

    def __len__(self):
        return self._total_transitions

5. World Dynamics Model Components



Following Dreamer V1, the World Dynamics model consists of four sub-networks:

| Component | Input | Output | Purpose |
|-----------|-------|--------|----------|
| Encoder | RGB image | Latent embedding | Compress visual observations |
| RSSM | Prev state + action | Next state (stochastic + deterministic) | Temporal dynamics |
| Reward Model | Latent state | Scalar reward | Predict rewards |
| Decoder | Latent state | Reconstructed image | Ensure latent captures visual info |

class Encoder(nn.Module):
    """CNN Encoder: maps RGB frames to a latent embedding.

    Uses a 4-layer CNN with ReLU activations. The output is flattened
    and projected to the latent dimension, capturing spatial features
    needed for the RSSM's state representation.

    Architecture: 4 conv layers with stride-2 downsampling,
    progressively doubling channels: depth -> 2*depth -> 4*depth -> 8*depth
    """

    def __init__(self, img_channels=IMG_CHANNELS, depth=ENCODER_DEPTH,
                 latent_dim=LATENT_DIM):
        super().__init__()
        self.convs = nn.Sequential(
            nn.Conv2d(img_channels, 1 * depth, 4, stride=2, padding=1),  # 64->32
            nn.ReLU(),
            nn.Conv2d(1 * depth, 2 * depth, 4, stride=2, padding=1),     # 32->16
            nn.ReLU(),
            nn.Conv2d(2 * depth, 4 * depth, 4, stride=2, padding=1),     # 16->8
            nn.ReLU(),
            nn.Conv2d(4 * depth, 8 * depth, 4, stride=2, padding=1),     # 8->4
            nn.ReLU(),
        )
        # 8*depth channels * 4*4 spatial = feature_dim
        self.feature_dim = 8 * depth * 4 * 4
        self.fc = nn.Linear(self.feature_dim, latent_dim)

    def forward(self, obs):
        """Encode observation to latent embedding.

        Args:
            obs: Image tensor (B, C, H, W) with values in [0, 1]

        Returns:
            Latent embedding tensor (B, latent_dim)
        """
        x = self.convs(obs)
        x = x.view(x.size(0), -1)  # Flatten spatial dimensions
        return self.fc(x)


class Decoder(nn.Module):
    """Transposed CNN Decoder: reconstructs frames from latent states.

    Mirrors the encoder architecture with transposed convolutions.
    Takes the combined latent state (deterministic + stochastic) and
    reconstructs the original RGB frame. The reconstruction loss ensures
    the latent state captures visually relevant information.
    """

    def __init__(self, state_dim=DETERMINISTIC_DIM + LATENT_DIM,
                 depth=ENCODER_DEPTH, img_channels=IMG_CHANNELS):
        super().__init__()
        self.feature_dim = 8 * depth * 4 * 4
        self.depth = depth

        self.fc = nn.Linear(state_dim, self.feature_dim)
        self.deconvs = nn.Sequential(
            nn.ConvTranspose2d(8 * depth, 4 * depth, 4, stride=2, padding=1),  # 4->8
            nn.ReLU(),
            nn.ConvTranspose2d(4 * depth, 2 * depth, 4, stride=2, padding=1),  # 8->16
            nn.ReLU(),
            nn.ConvTranspose2d(2 * depth, 1 * depth, 4, stride=2, padding=1),  # 16->32
            nn.ReLU(),
            nn.ConvTranspose2d(1 * depth, img_channels, 4, stride=2, padding=1),# 32->64
        )

    def forward(self, state):
        """Decode latent state to reconstructed frame.

        Args:
            state: Combined state (B, deterministic_dim + latent_dim)

        Returns:
            Reconstructed frame (B, C, H, W) with raw logits (pre-sigmoid)
        """
        x = self.fc(state)
        x = x.view(-1, 8 * self.depth, 4, 4)  # Reshape to spatial feature map
        return self.deconvs(x)

class RSSM(nn.Module):
    """Recurrent State-Space Model — the core of Dreamer's world model.

    Maintains two state components:
      - Deterministic state (h): GRU hidden state, captures long-term dynamics
      - Stochastic state (s): Gaussian latent, captures uncertainty

    Two pathways:
      - Prior: h_t, a_{t-1} -> h_{t+1} -> prior(s_{t+1})  (imagination)
      - Posterior: h_{t+1}, obs_{t+1} -> posterior(s_{t+1})  (training with real data)

    The KL divergence between prior and posterior drives the model to learn
    predictive dynamics that match observed transitions.
    """

    def __init__(self, latent_dim=LATENT_DIM, deterministic_dim=DETERMINISTIC_DIM,
                 hidden_dim=HIDDEN_DIM, action_dim=ACTION_DIM):
        super().__init__()
        self.latent_dim = latent_dim
        self.deterministic_dim = deterministic_dim

        # Pre-GRU: project (prev_stochastic + action) to GRU input size
        self.pre_gru = nn.Sequential(
            nn.Linear(latent_dim + action_dim, hidden_dim),
            nn.ReLU(),
        )

        # GRU cell for deterministic state transitions
        self.gru = nn.GRUCell(hidden_dim, deterministic_dim)

        # Prior network: deterministic state -> prior distribution parameters
        self.prior_net = nn.Sequential(
            nn.Linear(deterministic_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, 2 * latent_dim),  # mean + log_std
        )

        # Posterior network: deterministic state + encoded obs -> posterior params
        self.posterior_net = nn.Sequential(
            nn.Linear(deterministic_dim + latent_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, 2 * latent_dim),  # mean + log_std
        )

    def initial_state(self, batch_size):
        """Create zero-initialized RSSM state for the start of an episode."""
        return {
            'deterministic': torch.zeros(batch_size, self.deterministic_dim, device=device),
            'stochastic': torch.zeros(batch_size, self.latent_dim, device=device),
        }

    def observe_step(self, prev_state, action, encoded_obs):
        """Single RSSM step with observation (posterior pathway).

        Used during training when ground-truth observations are available.
        Returns both prior and posterior distributions for KL computation.

        Args:
            prev_state: Dict with 'deterministic' and 'stochastic' tensors
            action: Action tensor (B, action_dim)
            encoded_obs: Encoded observation from the Encoder (B, latent_dim)

        Returns:
            Tuple of (new_state, prior_dist, posterior_dist)
        """
        # Compute new deterministic state via GRU
        x = self.pre_gru(torch.cat([prev_state['stochastic'], action], dim=-1))
        deterministic = self.gru(x, prev_state['deterministic'])

        # Prior: predict stochastic state from deterministic alone
        prior_params = self.prior_net(deterministic)
        prior_mean, prior_log_std = prior_params.chunk(2, dim=-1)
        prior_std = F.softplus(prior_log_std) + 0.1  # Ensure positive std
        prior_dist = Normal(prior_mean, prior_std)

        # Posterior: incorporate observation information
        posterior_input = torch.cat([deterministic, encoded_obs], dim=-1)
        posterior_params = self.posterior_net(posterior_input)
        posterior_mean, posterior_log_std = posterior_params.chunk(2, dim=-1)
        posterior_std = F.softplus(posterior_log_std) + 0.1
        posterior_dist = Normal(posterior_mean, posterior_std)

        # Sample stochastic state from posterior (reparameterized)
        stochastic = posterior_dist.rsample()

        new_state = {
            'deterministic': deterministic,
            'stochastic': stochastic,
        }

        return new_state, prior_dist, posterior_dist

    def imagine_step(self, prev_state, action):
        """Single RSSM step without observation (prior pathway).

        Used for imagination rollouts during actor-critic training.
        Only uses the prior (no posterior correction from observations).

        Args:
            prev_state: Dict with 'deterministic' and 'stochastic' tensors
            action: Action tensor (B, action_dim)

        Returns:
            Tuple of (new_state, prior_dist)
        """
        x = self.pre_gru(torch.cat([prev_state['stochastic'], action], dim=-1))
        deterministic = self.gru(x, prev_state['deterministic'])

        prior_params = self.prior_net(deterministic)
        prior_mean, prior_log_std = prior_params.chunk(2, dim=-1)
        prior_std = F.softplus(prior_log_std) + 0.1
        prior_dist = Normal(prior_mean, prior_std)

        stochastic = prior_dist.rsample()

        new_state = {
            'deterministic': deterministic,
            'stochastic': stochastic,
        }

        return new_state, prior_dist

    def get_full_state(self, state):
        """Concatenate deterministic and stochastic parts into a single vector."""
        return torch.cat([state['deterministic'], state['stochastic']], dim=-1)

class Actor(nn.Module):
    """Policy network outputting continuous actions via a Gaussian distribution.

    Uses tanh squashing to bound actions to [-1, 1], following the SAC-style
    approach used in Dreamer V1. The log_std is state-dependent, allowing
    the policy to express different levels of certainty for different states.
    """

    def __init__(self, state_dim=DETERMINISTIC_DIM + LATENT_DIM,
                 hidden_dim=HIDDEN_DIM, action_dim=ACTION_DIM):
        super().__init__()
        self.trunk = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
        )
        self.mean_head = nn.Linear(hidden_dim, action_dim)
        self.log_std_head = nn.Linear(hidden_dim, action_dim)

    def forward(self, state):
        """Compute action distribution from latent state.

        Returns:
            Normal distribution over actions (before tanh squashing)
        """
        features = self.trunk(state)
        mean = self.mean_head(features)
        log_std = self.log_std_head(features)
        # Clamp log_std to prevent numerical instability
        log_std = torch.clamp(log_std, min=-5.0, max=2.0)
        std = torch.exp(log_std)
        return Normal(mean, std)

    def get_action(self, state, deterministic=False):
        """Sample an action, squashed through tanh to [-1, 1].

        Args:
            state: Combined latent state tensor
            deterministic: If True, return the mean (no sampling)

        Returns:
            Action tensor in [-1, 1]
        """
        dist = self.forward(state)
        if deterministic:
            raw_action = dist.mean
        else:
            raw_action = dist.rsample()
        return torch.tanh(raw_action)


class Critic(nn.Module):

    Trained via TD(lambda) using imagined trajectories from the world model.
    The value estimate V(s) is used to compute advantages for the actor.

                 hidden_dim=HIDDEN_DIM):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, 1),
        )

    def forward(self, state):
        """Estimate state value."""
        return self.net(state).squeeze(-1)

class WorldModel:
    """Complete Dreamer V1 World Model combining all sub-networks.

    Fixes from experiments:
      - FREE_NATS removed entirely so KL can train from the start
      - KL_WEIGHT reduced to 0.1 so reconstruction/reward get priority
      - Actor uses raw (unnormalized) advantage instead of normalized returns
      - Reward loss weighted higher to address growing reward error
    """

    def __init__(self):
        # World dynamics components
        self.encoder = Encoder().to(device)
        self.rssm = RSSM().to(device)
        self.reward_model = RewardModel().to(device)
        self.decoder = Decoder().to(device)

        # Actor-Critic components
        self.actor = Actor().to(device)
        self.critic = Critic().to(device)

        # Optimizer for world dynamics
        self.wm_params = (
            list(self.encoder.parameters()) +
            list(self.rssm.parameters()) +
            list(self.reward_model.parameters()) +
            list(self.decoder.parameters())
        )
        self.wm_optimizer = optim.Adam(self.wm_params, lr=LEARNING_RATE_WM, eps=1e-5)

        self.actor_optimizer = optim.Adam(
            self.actor.parameters(), lr=LEARNING_RATE_ACTOR, eps=1e-5
        )
        self.critic_optimizer = optim.Adam(
            self.critic.parameters(), lr=LEARNING_RATE_CRITIC, eps=1e-5
        )

        # Running value statistics for critic normalization only
        self._value_running_mean = 0.0
        self._value_running_var = 1.0
        self._value_momentum = 0.99

        # Training metrics
        self.wm_losses = []
        self.recon_losses = []
        self.kl_losses = []
        self.reward_losses = []
        self.actor_losses = []
        self.critic_losses = []

    def _update_value_stats(self, targets):
        """Update running mean/variance of value targets."""
        batch_mean = targets.mean().item()
        batch_var = targets.var().item()
        m = self._value_momentum
        self._value_running_mean = m * self._value_running_mean + (1 - m) * batch_mean
        self._value_running_var = m * self._value_running_var + (1 - m) * batch_var

    def _normalize_values(self, values):
        """Normalize values using running statistics."""
        std = max(self._value_running_var ** 0.5, 1e-6)
        return (values - self._value_running_mean) / std

    def train_world_model(self, batch):
        """Train the world dynamics model on a batch of sequences.

        Loss = recon + KL_WEIGHT * kl + 2.0 * reward_loss
        Reward loss is weighted 2x because it was underfit in previous runs.

        Args:
            batch: Dict from ReplayBuffer.sample_sequences()
        Returns:
            Dict of loss values for logging
        """
        frames = batch['frames']
        actions = batch['actions']
        rewards = batch['rewards']  # normalized

        B, T_plus_1, C, H, W = frames.shape
        T = T_plus_1 - 1

        state = self.rssm.initial_state(B)

        total_recon_loss = 0.0
        total_kl_loss = 0.0
        total_reward_loss = 0.0

        for t in range(T):
            encoded_obs = self.encoder(frames[:, t + 1])

            state, prior_dist, posterior_dist = self.rssm.observe_step(
                state, actions[:, t], encoded_obs
            )

            full_state = self.rssm.get_full_state(state)

            # Reconstruction loss
            recon = self.decoder(full_state)
            recon_loss = F.mse_loss(recon, frames[:, t + 1], reduction='mean')
            total_recon_loss += recon_loss

            # KL divergence — no free nats floor, trains freely from the start
            kl_div = torch.distributions.kl_divergence(posterior_dist, prior_dist)
            kl_div = kl_div.sum(dim=-1).mean()
            # Only apply free nats if > 0, otherwise use raw KL
            if FREE_NATS > 0:
                kl_loss = torch.max(kl_div, torch.tensor(FREE_NATS, device=device))
            else:
                kl_loss = kl_div
            total_kl_loss += kl_loss

            # Reward prediction loss
            predicted_reward = self.reward_model(full_state)
            reward_loss = F.mse_loss(predicted_reward, rewards[:, t])
            total_reward_loss += reward_loss

        total_recon_loss /= T
        total_kl_loss /= T
        total_reward_loss /= T

        # Reward loss weighted 2x to address underfitting
        wm_loss = total_recon_loss + KL_WEIGHT * total_kl_loss + 2.0 * total_reward_loss

        self.wm_optimizer.zero_grad()
        wm_loss.backward()
        nn.utils.clip_grad_norm_(self.wm_params, GRAD_CLIP)
        self.wm_optimizer.step()

        losses = {
            'total': wm_loss.item(),
            'recon': total_recon_loss.item(),
            'kl': total_kl_loss.item(),
            'reward': total_reward_loss.item(),
        }
        self.wm_losses.append(losses['total'])
        self.recon_losses.append(losses['recon'])
        self.kl_losses.append(losses['kl'])
        self.reward_losses.append(losses['reward'])

        return losses

    def train_actor_critic(self, batch):
        """Train actor and critic using imagined trajectories.

        Separate rollouts for critic and actor.
        Critic rollout is fully detached. Actor rollout has gradient flow
        through imagine_step and reward_model back to the actor.
        """
        frames = batch['frames']
        actions = batch['actions']
        B = frames.shape[0]

        # Get initial state from real observation (detached)
        with torch.no_grad():
            state = self.rssm.initial_state(B)
            encoded_obs = self.encoder(frames[:, 0])
            state, _, _ = self.rssm.observe_step(state, actions[:, 0], encoded_obs)

        init_state = {
            'deterministic': state['deterministic'].detach(),
            'stochastic': state['stochastic'].detach(),
        }

        # Critic phase: fully detached rollout for value regression
        with torch.no_grad():
            critic_states_list = []
            critic_rewards_list = []
            cs = {
                'deterministic': init_state['deterministic'].clone(),
                'stochastic': init_state['stochastic'].clone(),
            }
            for _ in range(IMAGINATION_HORIZON):
                full_s = self.rssm.get_full_state(cs)
                critic_states_list.append(full_s)
                action = self.actor.get_action(full_s)
                cs, _ = self.rssm.imagine_step(cs, action)
                r = self.reward_model(self.rssm.get_full_state(cs))
                critic_rewards_list.append(r)

            critic_states = torch.stack(critic_states_list, dim=0)
            critic_rewards = torch.stack(critic_rewards_list, dim=0)

        # Critic values (gradient only for critic params)
        critic_values = self.critic(critic_states)
        critic_targets = self._compute_lambda_targets(
            critic_rewards, critic_values.detach()
        )
        self._update_value_stats(critic_targets)

        normalized_targets = self._normalize_values(critic_targets)
        normalized_values = self._normalize_values(critic_values[:-1])
        critic_loss = F.mse_loss(normalized_values, normalized_targets.detach())

        self.critic_optimizer.zero_grad()
        critic_loss.backward()
        nn.utils.clip_grad_norm_(self.critic.parameters(), GRAD_CLIP)
        self.critic_optimizer.step()


        # Actor phase: fresh rollout with gradients flowing through
        # actor -> action -> rssm.imagine_step -> reward_model -> targets
        actor_rewards_list = []
        actor_states_list = []

        current_state = {
            'deterministic': init_state['deterministic'].clone(),
            'stochastic': init_state['stochastic'].clone(),
        }

        for _ in range(IMAGINATION_HORIZON):
            full_state = self.rssm.get_full_state(current_state)
            actor_states_list.append(full_state)

            # Actor gradient flows through this action
            action = self.actor.get_action(full_state)

            # World model used for dynamics (params not updated here)
            current_state, _ = self.rssm.imagine_step(current_state, action)
            reward = self.reward_model(self.rssm.get_full_state(current_state))
            actor_rewards_list.append(reward)

        actor_rewards = torch.stack(actor_rewards_list, dim=0)
        actor_states_t = torch.stack(actor_states_list, dim=0)

        # Critic as detached baseline
        with torch.no_grad():
            actor_values = self.critic(actor_states_t)

        actor_targets = self._compute_lambda_targets(
            actor_rewards, actor_values
        )

        # Direct value maximization
        # Normalize targets by their scale to keep actor gradients bounded.
        # Without this, actor_targets grow as the critic improves, causing actor loss to explode (experiment: -3 -> -44 -> -263).
        target_scale = max(actor_targets.abs().mean().item(), 1.0)
        actor_loss = -(actor_targets / target_scale).mean()

        self.actor_optimizer.zero_grad()
        actor_loss.backward()

        nn.utils.clip_grad_norm_(self.actor.parameters(), GRAD_CLIP)
        self.actor_optimizer.step()

        self.actor_losses.append(actor_loss.item())
        self.critic_losses.append(critic_loss.item())

        return {
            'actor_loss': actor_loss.item(),
            'critic_loss': critic_loss.item(),
        }
    def _compute_lambda_targets(self, rewards, values):
        """Compute TD(lambda) value targets."""
        H, B = rewards.shape
        targets = torch.zeros(H, B, device=device)
        targets[-1] = values[-1]

        for t in reversed(range(H - 1)):
            td_target = rewards[t] + GAMMA * values[t + 1]
            targets[t] = (
                (1 - LAMBDA_GAE) * td_target +
                LAMBDA_GAE * (rewards[t] + GAMMA * targets[t + 1])
            )

        return targets[:-1]

def collect_episode(env, world_model=None, random_policy=False):
    """Collect one full episode of transitions.

    If random_policy=True or no world_model is given, actions are sampled
    uniformly at random. Otherwise, the actor selects actions based on
    the current RSSM state.

    Args:
        env: Gymnasium environment
        world_model: Optional WorldModel for policy-based collection
        random_policy: Force random actions if True

    Returns:
        Tuple of (frames, actions, rewards, dones, total_reward)
    """
    obs, info = env.reset()
    frame = preprocess_frame(env.render())
    frames = [frame]
    actions = []
    rewards = []
    dones = []
    total_reward = 0.0

    # Initialize RSSM state if using learned policy
    if world_model and not random_policy:
        state = world_model.rssm.initial_state(1)

    done = False
    truncated = False

    while not (done or truncated):
        if random_policy or world_model is None:
            # Random action in [-1, 1] for both steering and acceleration
            action = env.action_space.sample()
        else:
            # Use learned policy
            with torch.no_grad():
                frame_tensor = frame.unsqueeze(0).to(device)
                encoded = world_model.encoder(frame_tensor)
                action_tensor = torch.FloatTensor(actions[-1] if actions else [0.0, 0.0]).unsqueeze(0).to(device)
                state, _, _ = world_model.rssm.observe_step(state, action_tensor, encoded)
                full_state = world_model.rssm.get_full_state(state)
                action = world_model.actor.get_action(full_state).cpu().numpy().squeeze()

        # Ensure action is numpy array
        if isinstance(action, torch.Tensor):
            action = action.cpu().numpy()
        action = np.array(action, dtype=np.float32)

        obs, reward, done, truncated, info = env.step(action)
        frame = preprocess_frame(env.render())

        frames.append(frame)
        actions.append(action)
        rewards.append(float(reward))
        dones.append(float(done or truncated))
        total_reward += reward

    return frames, actions, rewards, dones, total_reward

def visualize_predictions(world_model, env, n_steps=PREDICTION_HORIZON):
    """Generate side-by-side visualization of predicted vs actual frames.

    Collects a real trajectory and simultaneously generates an imagined
    trajectory starting from the same initial state. Plots frames at
    regular intervals for visual comparison.

    Args:
        world_model: Trained WorldModel instance
        env: Gymnasium environment
        n_steps: Number of prediction steps
    """
    obs, info = env.reset(seed=SEED + 100)
    initial_frame = preprocess_frame(env.render())

    # Collect actual trajectory
    actual_frames = [initial_frame]
    actions_taken = []
    actual_rewards = []

    with torch.no_grad():
        # Initialize RSSM state
        state = world_model.rssm.initial_state(1)
        frame_tensor = initial_frame.unsqueeze(0).to(device)
        encoded = world_model.encoder(frame_tensor)

        # Use zero action for first step encoding
        zero_action = torch.zeros(1, ACTION_DIM, device=device)
        state, _, _ = world_model.rssm.observe_step(state, zero_action, encoded)

        # Save initial state for imagination rollout
        imagine_state = {
            'deterministic': state['deterministic'].clone(),
            'stochastic': state['stochastic'].clone(),
        }

        # Collect actual trajectory and imagined predictions
        predicted_frames = [initial_frame]  # First frame is shared
        predicted_rewards = []

        for step in range(n_steps):
            # Get action from policy
            full_state = world_model.rssm.get_full_state(imagine_state)
            action = world_model.actor.get_action(full_state, deterministic=True)
            action_np = action.cpu().numpy().squeeze()
            actions_taken.append(action_np)

            # Actual environment step
            obs, reward, done, truncated, info = env.step(action_np)
            actual_frame = preprocess_frame(env.render())
            actual_frames.append(actual_frame)
            actual_rewards.append(reward)

            if done or truncated:
                break

            # Imagined step (prior only — no observation)
            imagine_state, _ = world_model.rssm.imagine_step(imagine_state, action)
            full_state_pred = world_model.rssm.get_full_state(imagine_state)

            # Decode predicted frame
            decoded = world_model.decoder(full_state_pred)
            decoded_sq = decoded.squeeze(0).cpu()
            d_min = decoded_sq.min()
            d_max = decoded_sq.max()
            if d_max - d_min > 1e-6:
                pred_frame = (decoded_sq - d_min) / (d_max - d_min)
            else:
                pred_frame = decoded_sq * 0.0

            #
            predicted_frames.append(pred_frame)

            # Predicted reward
            pred_reward = world_model.reward_model(full_state_pred).item()
            predicted_rewards.append(pred_reward)

    # Select frames at regular intervals for display
    num_display = min(8, len(actual_frames))
    indices = np.linspace(0, len(actual_frames) - 1, num_display, dtype=int)

    fig, axes = plt.subplots(2, num_display, figsize=(3 * num_display, 6))
    fig.suptitle('N-Step Prediction: Actual vs. Imagined',
                 fontsize=14, fontweight='bold')

    for col, idx in enumerate(indices):
        # Actual frame
        axes[0, col].imshow(actual_frames[idx].permute(1, 2, 0).numpy().clip(0, 1))
        axes[0, col].set_title(f't={idx}', fontsize=10)
        axes[0, col].axis('off')
        if col == 0:
            axes[0, col].set_ylabel('Actual', fontsize=12, fontweight='bold')

        # Predicted frame
        if idx < len(predicted_frames):
            axes[1, col].imshow(predicted_frames[idx].permute(1, 2, 0).numpy().clip(0, 1))
        axes[1, col].set_title(f't={idx}', fontsize=10)
        axes[1, col].axis('off')
        if col == 0:
            axes[1, col].set_ylabel('Predicted', fontsize=12, fontweight='bold')

    plt.tight_layout()
    plt.savefig('prediction_comparison.png', dpi=150, bbox_inches='tight')
    plt.show()

    # Plot reward comparison
    if predicted_rewards:
        fig, ax = plt.subplots(figsize=(10, 4))
        steps = range(len(actual_rewards[:len(predicted_rewards)]))
        ax.plot(steps, actual_rewards[:len(predicted_rewards)],
                'b-o', markersize=3, label='Actual Reward', linewidth=1.5)
        ax.plot(steps, predicted_rewards,
                'r--s', markersize=3, label='Predicted Reward', linewidth=1.5)
        ax.set_xlabel('Step')
        ax.set_ylabel('Reward')
        ax.set_title('Reward Prediction: Actual vs. Imagined')
        ax.legend()
        ax.grid(True, alpha=0.3)
        plt.tight_layout()
        plt.savefig('reward_prediction.png', dpi=150, bbox_inches='tight')
        plt.show()

    return actual_frames, predicted_frames, actual_rewards, predicted_rewards

def create_comparison_video(actual_frames, predicted_frames,
                            filename='dreamer_comparison.mp4', fps=5):
    """Create a side-by-side video comparing actual vs predicted frames.

    Produces an MP4 video with actual frames on the left and predicted
    frames on the right, with labels overlaid.

    Args:
        actual_frames: List of actual frame tensors (C, H, W)
        predicted_frames: List of predicted frame tensors (C, H, W)
        filename: Output filename
        fps: Frames per second
    """
    import cv2

    n_frames = min(len(actual_frames), len(predicted_frames))
    video_frames = []

    for i in range(n_frames):
        # Convert tensors to numpy arrays (HWC, uint8)
        actual_np = (actual_frames[i].permute(1, 2, 0).numpy().clip(0, 1) * 255).astype(np.uint8)
        pred_raw = predicted_frames[i].permute(1, 2, 0).numpy()
        p_min, p_max = pred_raw.min(), pred_raw.max()
        if p_max - p_min > 1e-6:
            pred_np = ((pred_raw - p_min) / (p_max - p_min) * 255).astype(np.uint8)
        else:
            pred_np = np.zeros_like(pred_raw, dtype=np.uint8)
        #

        # Upscale for visibility
        scale = 4
        actual_up = cv2.resize(actual_np, (IMG_WIDTH * scale, IMG_HEIGHT * scale),
                                interpolation=cv2.INTER_NEAREST)
        pred_up = cv2.resize(pred_np, (IMG_WIDTH * scale, IMG_HEIGHT * scale),
                              interpolation=cv2.INTER_NEAREST)

        # Add labels
        cv2.putText(actual_up, 'ACTUAL', (10, 25),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.8, (255, 255, 255), 2)
        cv2.putText(pred_up, 'PREDICTED', (10, 25),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.8, (255, 255, 255), 2)
        cv2.putText(actual_up, f't={i}', (10, 55),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 0), 2)
        cv2.putText(pred_up, f't={i}', (10, 55),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 0), 2)

        # Separator bar
        separator = np.ones((IMG_HEIGHT * scale, 4, 3), dtype=np.uint8) * 255

        # Concatenate side by side
        combined = np.concatenate([actual_up, separator, pred_up], axis=1)
        video_frames.append(combined)

    # Save video
    imageio.mimsave(filename, video_frames, fps=fps)
    print(f"Video saved: {filename} ({n_frames} frames at {fps} FPS)")
    return filename

def evaluate_prediction_quality(world_model, n_steps=30):
    """Measure prediction quality as a function of prediction horizon.

    For each step t, computes the MSE between the world model's predicted
    frame at step t and the actual frame at step t.

    Returns:
        List of MSE values indexed by prediction step
    """
    eval_env = make_env()
    obs, info = eval_env.reset(seed=SEED + 300)
    initial_frame = preprocess_frame(eval_env.render())

    mse_per_step = []

    with torch.no_grad():
        state = world_model.rssm.initial_state(1)
        frame_tensor = initial_frame.unsqueeze(0).to(device)
        encoded = world_model.encoder(frame_tensor)
        zero_action = torch.zeros(1, ACTION_DIM, device=device)
        state, _, _ = world_model.rssm.observe_step(state, zero_action, encoded)

        imagine_state = {
            'deterministic': state['deterministic'].clone(),
            'stochastic': state['stochastic'].clone(),
        }

        for step in range(n_steps):
            full_state = world_model.rssm.get_full_state(imagine_state)
            action = world_model.actor.get_action(full_state, deterministic=True)
            action_np = action.cpu().numpy().squeeze()

            # Actual step
            obs, reward, done, truncated, info = eval_env.step(action_np)
            actual_frame = preprocess_frame(eval_env.render()).to(device)

            if done or truncated:
                break

            # Imagined step
            imagine_state, _ = world_model.rssm.imagine_step(imagine_state, action)
            pred_full = world_model.rssm.get_full_state(imagine_state)
            decoded = torch.sigmoid(world_model.decoder(pred_full)).squeeze(0)

            # Compute MSE between predicted and actual frame
            mse = F.mse_loss(decoded, actual_frame).item()
            mse_per_step.append(mse)

    eval_env.close()
    return mse_per_step
